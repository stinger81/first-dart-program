num? calculate(num n1, String operation,[num n2=0]) {
  switch (operation) {
    case '*':
      return multiply(n1, n2);
    case '/':
      return divide(n1, n2);
    case '+':
      return add(n1, n2);
    case '-':
      return subtract(n1, n2);
    case '^':
      return exponent(n1, n2);
    case '%':
      return modulo(n1, n2);
    case 'sqrt':
      return squareRoot(n1);
    default:
      return null;
  }
}
